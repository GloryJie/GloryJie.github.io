            // ËæπÊñúÁöÑÊÉÖÂÜµ
            if (target.getLng() <= minLng) {
                // ÁÇπpointÁöÑxÂùêÊ†áÂú®ËæπÁöÑÂ∑¶‰æßÔºåÈÇ£‰πàËÇØÂÆöÁõ∏‰∫§
                intersectPointCount++;
            } else if (target.getLng() < maxLng) {
                // ÁÇπpointÁöÑxÂùêÊ†áÂú®ÁÇπB„ÄÅAÁöÑxÂùêÊ†á‰∏≠Èó¥
                double diff = (target.getLat() - pointA.getLat()) * (pointA.getLng() - pointB.getLng()) / (pointA.getLat() - pointB.getLat()) + pointA.getLng() - target.getLng();
                if (diff >= 0) {
                    if (diff < PRECISION) {
                        // Áî±‰∫édoubleÁ≤æÂ∫¶Âú®ËÆ°ÁÆóÊó∂‰ºöÊúâÊçüÂ§±ÔºåÊïÖÂåπÈÖç‰∏ÄÂÆöÁöÑÂÆπÂ∑Æ
                        // ÁÇπÂú®ÊñúÁ∫øP1P2‰∏ä
                        return true;
                    } else {
                        // ÁÇπÂêëÂè≥Â∞ÑÁ∫ø‰∏éÊñúÁ∫øP1P2Êúâ‰∫§ÁÇπ
                        intersectPointCount++;
                    }
                }
            }
            pointA = pointB;
        }
        return (intersectPointCount + Math.abs(intersectPointWeights)) % 2 != 0;
    }
